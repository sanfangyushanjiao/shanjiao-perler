```typescript
// 检查库存并自动补货
const checkAndRefill = async () => {
  const { data: stock } = await supabase
    .from('activation_codes')
    .select('type, status')
    .eq('status', 'unused');

  const counts = {
    '24h': stock.filter(s => s.type === '24h').length,
    '7d': stock.filter(s => s.type === '7d').length,
    'lifetime': stock.filter(s => s.type === 'lifetime').length,
  };

  // 如果库存不足，自动生成
  if (counts['24h'] < 50) {
```
